```python
import git
import time

# 初始化 Git 仓库
repo = git.Repo('/path/to/your/repository')

# 提交函数
def commit_to_git():
    index = repo.index
    index.add('*')
    index.commit('提交说明')

# 设置计时器
while True:
    commit_to_git()
    time.sleep(30 * 60)  # 等待 30 分钟

```
